Suddenly JavaScript can understand

var bookId = 'http://example.net/b/wikinomics';
var v = new VIE();
v.use(new v.RdfaService());
v.load({ element: 'body'}).
  using('rdfa').
  execute().
  done(function () {
    var myBook = v.entities.get(bookId);
    alert(myBook.get('dc:title'));
    // Says "Wikinomics"
  });
