Using the Dialog component

// Import the dialog component
var Dialog = require('dialog').Dialog;

// Initialize a dialog with the options we wanted
var dialog = new Dialog({
  title: 'Hello, world!',
  actions: {
    'Yes': function () { alert("CLICKED YES"); },
    'No': function () { alert("CLICKED NO"); }
  }
});

// Render and place to DOM
container.appendChild(dialog.render().el);
